function clean_input($data) {
    return htmlspecialchars(trim($data), ENT_QUOTES, 'UTF-8');
}

function read_file_lines($file_tmp) {
    $lines = [];
    if (is_uploaded_file($file_tmp) && $_FILES['file']['size'] <= MAX_FILE_SIZE) {
        $content = file_get_contents($file_tmp);
        $content = str_replace(["\r\n", "\r"], "\n", $content);
        $lines = explode("\n", $content);
        $lines = array_map('trim', $lines);
        $lines = array_filter($lines, function($line) {
            return $line !== '';
        });
    }
    return array_values($lines);
}

function create_folder($path) {
    if (!is_dir($path)) {
        mkdir($path, 0755, true);
    }
}

function replace_placeholders($template, $data) {
    $search = [];
    $replace = [];
    foreach ($data as $key => $value) {
        $search[] = '{' . $key . '}';
        $replace[] = $value;
    }
    return str_replace($search, $replace, $template);
}

function generate_sitemap($urls, $base_url) {
    $xml = '<?xml version="1.0" encoding="UTF-8"?>' . "\n";
    $xml .= '<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">' . "\n";
    foreach ($urls as $url) {
        $xml .= '  <url><loc>' . htmlspecialchars($url) . '</loc></url>' . "\n";
    }
    $xml .= '</urlset>';
    return $xml;
}

// Fungsi crawling internal dengan depth control
function crawl_domain($start_url, $max_pages = 200, $depth = 3) {
    $parsed = parse_url($start_url);
    $base_domain = $parsed['scheme'] . '://' . $parsed['host'];
    $visited = [];
    $to_visit = [[$start_url, 0]];
    $found_urls = [];

    while (!empty($to_visit) && count($found_urls) < $max_pages) {
        list($url, $current_depth) = array_shift($to_visit);
        if (in_array($url, $visited) || $current_depth > $depth) continue;
        $visited[] = $url;

        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
        curl_setopt($ch, CURLOPT_USERAGENT, 'Mozilla/5.0 (compatible; NHUltraBot/2.0)');
        curl_setopt($ch, CURLOPT_TIMEOUT, CRAWL_TIMEOUT);
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
        $html = curl_exec($ch);
        $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);

        if ($httpCode == 200 && !empty($html)) {
            $found_urls[] = $url;
            if ($current_depth < $depth) {
                preg_match_all('/<a[^>]+href="([^"]+)"/i', $html, $matches);
                foreach ($matches[1] as $link) {
                    $abs = url_to_absolute($link, $url);
                    if ($abs && strpos($abs, $base_domain) === 0 && !in_array($abs, $visited)) {
                        $to_visit[] = [$abs, $current_depth + 1];
                    }
                }
            }
        }
    }
    return $found_urls;
}

function url_to_absolute($relative, $base) {
    if (parse_url($relative, PHP_URL_SCHEME) != '') return $relative;
    if ($relative[0] == '#') return false;
    if ($relative[0] == '/') {
        $parts = parse_url($base);
        return $parts['scheme'] . '://' . $parts['host'] . $relative;
    }
    $base = rtrim($base, '/') . '/';
    return $base . ltrim($relative, '/');
}

// Fungsi Google Custom Search API
function google_custom_search($domain, $api_key, $cx, $max_results = 100) {
    $results = [];
    $start = 1;
    while ($start <= $max_results) {
        $url = "https://www.googleapis.com/customsearch/v1?key={$api_key}&cx={$cx}&q=site:{$domain}&num=10&start={$start}";
        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_TIMEOUT, 30);
        $response = curl_exec($ch);
        $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);

        if ($httpCode == 200) {
            $data = json_decode($response, true);
            if (isset($data['items'])) {
                foreach ($data['items'] as $item) {
                    $results[] = $item['link'];
                }
            }
            if (!isset($data['queries']['nextPage'])) break;
            $start += 10;
        } else {
            break;
        }
    }
    return $results;
}

if (isset($_POST['scan_submit'])) {
    $domain = clean_input($_POST['domain']);
    $method = $_POST['scan_method'] ?? 'crawl';
    $depth = intval($_POST['depth'] ?? 3);
    $export_format = $_POST['export_format'] ?? 'txt';
    $api_key = clean_input($_POST['api_key'] ?? '');
    $cx = clean_input($_POST['cx'] ?? '');

    if (!empty($domain)) {
        // Untuk node dork, kita tidak butuh parsing URL
        if ($method != 'node_dork') {
            if (!preg_match('/^https?:\/\//', $domain)) {
                $domain = 'http://' . $domain;
            }
            $parsed = parse_url($domain);
            $host = $parsed['host'] ?? '';
        } else {
            $host = $domain; // Bypass host check
        }

        if ($host || $method == 'node_dork') {
            if ($method == 'google_api' && !empty($api_key) && !empty($cx)) {
                $links = google_custom_search($host, $api_key, $cx, 100);
                $method_used = "Google Custom Search API";
            } elseif ($method == 'node_dork') {
                // Dorking ditangani via AJAX di JavaScript.
                // Jika sampai sini berarti JavaScript tidak jalan (browser lama).
                $scan_message = "Mode Google Dorking membutuhkan JavaScript aktif di browser.";
                $method_used = "Python Google Dorker API";
            } else {
                $links = crawl_domain($domain, MAX_CRAWL_PAGES, $depth);
                $method_used = "Crawling internal (depth $depth, maks " . MAX_CRAWL_PAGES . " halaman)";
            }

            if (!isset($links)) $links = [];

            if (empty($links) && !isset($scan_message)) {
                $scan_message = "Tidak ditemukan halaman. Metode: $method_used. Coba metode lain atau periksa domain.";
            } elseif (!empty($links)) {
                if ($export_format == 'csv') {
                    $file_content = implode("\n", $links);
                    $filename = 'scan_result.csv';
                } elseif ($export_format == 'json') {
                    $file_content = json_encode($links, JSON_PRETTY_PRINT);
                    $filename = 'scan_result.json';
                } else {
                    $file_content = implode("\n", $links);
                    $filename = SCAN_RESULT_FILE;
                }
                file_put_contents($filename, $file_content);
                $scan_message = "Ditemukan " . count($links) . " URL menggunakan metode <strong>$method_used</strong>. <a href='$filename' download>Download hasil ($export_format)</a>";
            }
        } else {
            $scan_message = "Domain tidak valid.";
        }
    } else {
        $scan_message = "Masukkan domain.";
    }
}

// Proses Generator - dengan link daftar dari input teks
if (isset($_POST['generate_submit'])) {
    $base_canonical = clean_input($_POST['base_canonical']);
    $mode = $_POST['mode'] ?? 'roundrobin';
    $generate_sitemap = isset($_POST['generate_sitemap']);
    $sitemap_url = clean_input($_POST['sitemap_url'] ?? $base_canonical);

    if (!preg_match('/^https?:\/\//', $base_canonical)) {
        $base_canonical = 'http://' . $base_canonical;
    }
    $base_canonical = rtrim($base_canonical, '/') . '/';

    // Baca link daftar dari input teks
    $linkdaftar = [];
    for ($i = 1; $i <= 10; $i++) {
        $link = clean_input($_POST['linkdaftar' . $i] ?? '');
        // Jika diisi, simpan sebagai array (1 elemen) agar kompatibel dengan logika roundrobin/random
        // Tapi karena hanya satu link per tempat, kita bisa langsung gunakan string.
        // Untuk kemudahan, kita jadikan array dengan satu elemen jika tidak kosong.
        if (!empty($link)) {
            $linkdaftar[$i] = [$link];
        } else {
            $linkdaftar[$i] = [];
        }
    }

    // Upload file lainnya
    $folder_file = $_FILES['folder_file'] ?? null;
    $title_file = $_FILES['title_file'] ?? null;
    $brand_file = $_FILES['brand_file'] ?? null;
    $description_file = $_FILES['description_file'] ?? null;
    $gambar_file = $_FILES['gambar_file'] ?? null;
    
    $template_files = [];
    for ($i = 1; $i <= 10; $i++) {
        if (isset($_FILES['template_' . $i]) && $_FILES['template_' . $i]['error'] == UPLOAD_ERR_OK) {
            $template_files[] = $_FILES['template_' . $i];
        }
    }
    
    if (!$folder_file || $folder_file['error'] != UPLOAD_ERR_OK) {
        $generate_message = "File folder.txt wajib diupload.";
    } elseif (empty($template_files)) {
        $generate_message = "Minimal satu file template harus diupload.";
    } else {
        $folders = read_file_lines($folder_file['tmp_name']);
        $titles = $title_file && $title_file['error'] == UPLOAD_ERR_OK ? read_file_lines($title_file['tmp_name']) : [];
        $brands = $brand_file && $brand_file['error'] == UPLOAD_ERR_OK ? read_file_lines($brand_file['tmp_name']) : [];
        $descriptions = $description_file && $description_file['error'] == UPLOAD_ERR_OK ? read_file_lines($description_file['tmp_name']) : [];
        $gambars = $gambar_file && $gambar_file['error'] == UPLOAD_ERR_OK ? read_file_lines($gambar_file['tmp_name']) : [];
        
        $templates = [];
        foreach ($template_files as $tf) {
            $content = file_get_contents($tf['tmp_name']);
            if (!empty($content)) {
                $templates[] = $content;
            }
        }
        
        $total_folders = count($folders);
        $total_templates = count($templates);
        $total_titles = count($titles);
        $total_brands = count($brands);
        $total_descs = count($descriptions);
        $total_gambars = count($gambars);
        
        $generated_count = 0;
        $generated_urls = [];
        
        for ($idx = 0; $idx < $total_folders; $idx++) {
            $folder_path = trim($folders[$idx]);
            if (empty($folder_path)) continue;
            
            if ($mode == 'random') {
                $title = $total_titles > 0 ? $titles[array_rand($titles)] : '';
                $brand = $total_brands > 0 ? $brands[array_rand($brands)] : '';
                $desc = $total_descs > 0 ? $descriptions[array_rand($descriptions)] : '';
                $gambar = $total_gambars > 0 ? $gambars[array_rand($gambars)] : '';
                $template = $templates[array_rand($templates)];
            } else {
                $title = $total_titles > 0 ? $titles[$idx % $total_titles] : '';
                $brand = $total_brands > 0 ? $brands[$idx % $total_brands] : '';
                $desc = $total_descs > 0 ? $descriptions[$idx % $total_descs] : '';
                $gambar = $total_gambars > 0 ? $gambars[$idx % $total_gambars] : '';
                $template = $templates[$idx % $total_templates];
            }
            
            $canonical_url = $base_canonical . $folder_path . '/';
            $generated_urls[] = $canonical_url;
            
            // Generate random number for {random-number}
            $random_number = rand(100000000, 999999999);
            
            $data = [
                'title' => $title,
                'brand' => $brand,
                'description' => $desc,
                'gambar' => $gambar,
                'canonical' => $canonical_url,
                'random-number' => $random_number,
            ];
            
            // Link daftar: jika array tidak kosong, ambil berdasarkan mode; jika kosong, string kosong
            for ($l = 1; $l <= 10; $l++) {
                if (!empty($linkdaftar[$l])) {
                    // Karena hanya satu link per tempat, kita bisa langsung ambil elemen pertama
                    // Tapi untuk konsistensi dengan mode random, kita perlakukan seperti array
                    if ($mode == 'random') {
                        $data['linkdaftar' . $l] = $linkdaftar[$l][0]; // hanya satu, jadi ambil itu
                    } else {
                        $data['linkdaftar' . $l] = $linkdaftar[$l][0];
                    }
                } else {
                    $data['linkdaftar' . $l] = '';
                }
            }
            
            $content = replace_placeholders($template, $data);
            
            $full_path = BASE_PATH . '/' . $folder_path;
            create_folder($full_path);
            file_put_contents($full_path . '/index.php', $content);
            $generated_count++;
        }
        
        if ($generate_sitemap && !empty($generated_urls)) {
            $sitemap_content = generate_sitemap($generated_urls, $sitemap_url);
            file_put_contents(BASE_PATH . '/' . SITEMAP_FILE, $sitemap_content);
            $sitemap_msg = " Sitemap juga telah dibuat: <a href='" . SITEMAP_FILE . "' download>Download sitemap.xml</a>";
        } else {
            $sitemap_msg = '';
        }
        
        $generate_message = "Berhasil generate $generated_count halaman landing. $sitemap_msg";
    }
}

// Proses AMP Generator dengan fitur baru
if (isset($_POST['amp_submit'])) {
    $amp_template_file = $_FILES['amp_template'] ?? null;
    $amp_template = '';
    if ($amp_template_file && $amp_template_file['error'] == UPLOAD_ERR_OK) {
        $amp_template = file_get_contents($amp_template_file['tmp_name']);
    }
    $selected_folders = $_POST['selected_folders'] ?? [];
    $base_canonical_amp = clean_input($_POST['base_canonical_amp'] ?? '');
    
    if (!preg_match('/^https?:\/\//', $base_canonical_amp)) {
        $base_canonical_amp = 'http://' . $base_canonical_amp;
    }
    $base_canonical_amp = rtrim($base_canonical_amp, '/') . '/';
    
    // Scan SEMUA folder (termasuk nested) yang punya index.php
    $folders = [];
    $rii = new RecursiveIteratorIterator(
        new RecursiveDirectoryIterator(BASE_PATH, RecursiveDirectoryIterator::SKIP_DOTS),
        RecursiveIteratorIterator::SELF_FIRST
    );
    foreach ($rii as $file) {
        if ($file->isDir()) continue;
        if ($file->getFilename() !== 'index.php') continue;
        $dir = dirname($file->getPathname());
        $rel = ltrim(str_replace(realpath(BASE_PATH), '', realpath($dir)), DIRECTORY_SEPARATOR);
        $rel = str_replace('\\', '/', $rel);
        // Skip folder amp itu sendiri, dan skip root
        if ($rel === '' || $rel === AMP_FOLDER_NAME) continue;
        if (preg_match('#(^|/)' . preg_quote(AMP_FOLDER_NAME, '#') . '(/|$)#', $rel)) continue;
        $folders[] = $rel;
    }
    
    $amp_count = 0;
    foreach ($folders as $folder) {
        if (!empty($selected_folders) && !in_array($folder, $selected_folders)) continue;
        
        $source_index = BASE_PATH . '/' . $folder . '/index.php';
        $amp_folder = BASE_PATH . '/' . $folder . '/' . AMP_FOLDER_NAME;
        create_folder($amp_folder);
        $target_index = $amp_folder . '/index.php';
        
        // Canonical: halaman utama = base + folder path, AMP = base + folder path + /amp/
        $canonical_main = $base_canonical_amp . $folder . '/';
        $canonical_amp = $base_canonical_amp . $folder . '/' . AMP_FOLDER_NAME . '/';
        
        // Buat konten AMP
        if (!empty($amp_template)) {
            // Template AMP: {canonical} = URL halaman utama (bukan AMP sendiri)
            $amp_content = str_replace('{canonical}', $canonical_main, $amp_template);
        } else {
            // Jika tidak ada template AMP, gunakan konten asli dengan mengganti canonical
            if (file_exists($source_index)) {
                $original_content = file_get_contents($source_index);
                $amp_content = str_replace('{canonical}', $canonical_main, $original_content);
            } else {
                continue;
            }
        }
        
        file_put_contents($target_index, $amp_content);
        
        // Sekarang update file index.php utama: ganti {amp} dengan URL AMP
        if (file_exists($source_index)) {
            $main_content = file_get_contents($source_index);
            // Ganti placeholder {amp} dengan URL AMP (jika ada)
            $main_content = str_replace('{amp}', $canonical_amp, $main_content);
            file_put_contents($source_index, $main_content);
        }
        
        $amp_count++;
    }
    
    $amp_message = "Berhasil membuat AMP untuk $amp_count folder dan memperbarui placeholder {amp} di halaman utama.";
}

        const fontSize = 16;

        function initMatrix() {
            width = window.innerWidth;
            height = window.innerHeight;
            canvas.width = width;
            canvas.height = height;
            columns = [];
            const colCount = Math.floor(width / fontSize);
            for (let i = 0; i < colCount; i++) {
                columns[i] = Math.floor(Math.random() * height / fontSize);

        function drawMatrix() {
            ctx.fillStyle = 'rgba(0, 0, 0, 0.05)';
            ctx.fillRect(0, 0, width, height);
            ctx.fillStyle = '#0f0';
            ctx.font = fontSize + 'px monospace';
            for (let i = 0; i < columns.length; i++) {
                const char = chars[Math.floor(Math.random() * chars.length)];
                ctx.fillText(char, i * fontSize, columns[i] * fontSize);
                if (columns[i] * fontSize > height && Math.random() > 0.975) {
                    columns[i] = 0;
                } else {
                    columns[i]++;
                }
                if (Math.random() > 0.98) {
                    ctx.fillStyle = '#fff';
                    ctx.fillText(char, i * fontSize, columns[i] * fontSize);
                    ctx.fillStyle = '#0f0';
                }
            }
            requestAnimationFrame(drawMatrix);
        }
